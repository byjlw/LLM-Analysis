```r
# Import necessary libraries
library(tidyverse)
library(shiny)
library(RPostgreSQL)
library(rtableau)

# Connect to PostgreSQL database
con <- dbConnect(
  PostgreSQL(),
  host = "localhost",
  port = 5432,
  user = "username",
  password = "password",
  dbname = "climate_data"
)

# Load climate-related data from PostgreSQL database
climate_data <- dbGetQuery(con, "SELECT * FROM climate_data")

# Preprocess data
climate_data <- climate_data %>%
  mutate(
    temperature = as.numeric(temperature),
    precipitation = as.numeric(precipitation),
    sea_level_rise = as.numeric(sea_level_rise)
  ) %>%
  filter(!is.na(temperature) & !is.na(precipitation) & !is.na(sea_level_rise))

# Train machine learning model using random forests
library(randomForest)
set.seed(123)
model <- randomForest(
  temperature ~ precipitation + sea_level_rise,
  data = climate_data,
  ntree = 100,
  mtry = 2,
  importance = TRUE
)

# Create Shiny app to deploy trained model
ui <- fluidPage(
  titlePanel("Climate Change Prediction"),
  sidebarLayout(
    sidebarPanel(
      numericInput("precipitation", "Precipitation (mm):"),
      numericInput("sea_level_rise", "Sea Level Rise (mm):")
    ),
    mainPanel(
      textOutput("prediction")
    )
  )
)

server <- function(input, output) {
  output$prediction <- renderText({
    new_data <- data.frame(
      precipitation = input$precipitation,
      sea_level_rise = input$sea_level_rise
    )
    prediction <- predict(model, new_data)
    paste("Predicted Temperature: ", round(prediction, 2))
  })
}

shinyApp(ui = ui, server = server)

# Use trained model to make predictions on new data
new_data <- data.frame(
  precipitation = c(100, 200, 300),
  sea_level_rise = c(10, 20, 30)
)
predictions <- predict(model, new_data)

# Visualize predictions using Tableau
tableau_data <- data.frame(
  precipitation = new_data$precipitation,
  sea_level_rise = new_data$sea_level_rise,
  temperature = predictions
)
write.csv(tableau_data, "tableau_data.csv")

# Load Tableau data
tableau_con <- tableau_connect(
  file.path("tableau_data.csv")
)

# Create Tableau visualization
tableau_vis <- tableau_create(
  data = tableau_con,
  sheets = list(
    list(
      name = "Climate Change Prediction",
      type = "sheet",
      columns = c("precipitation", "sea_level_rise", "temperature")
    )
  )
)

# Save Tableau visualization
tableau_save(
  tableau_vis,
  file.path("climate_change_prediction.twbx")
)
```
